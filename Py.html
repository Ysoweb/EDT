أساسيات بايثون

تعلم أساسيات لغة بايثون مع شرح مفصل للأكواد

1. المتغيرات

# مثال على المتغيرات في بايثون
message = "مرحبا بالعالم!"  # نص
number = 42  # عدد صحيح

المتغيرات هي أماكن لتخزين البيانات في الذاكرة.

2. الجمل الشرطية

# مثال على الجمل الشرطية في بايثون
if number > 20:
    print("العدد أكبر من 20")
else:
    print("العدد 20 أو أقل")

تُستخدم الجمل الشرطية لاتخاذ قرارات بناءً على شروط معينة.

3. الحلقات التكرارية

# مثال على الحلقات التكرارية في بايثون
for i in range(5):
    print("العدد هو:", i)

تُستخدم الحلقات التكرارية لتكرار مجموعة من الأوامر عدة مرات.

4. الدوال

# مثال على الدوال في بايثون
def greet(name):
    return "مرحبا، " + name + "!"

print(greet("أحمد"))

الدوال هي مجموعة من الأوامر التي يمكن استخدامها عدة مرات في أماكن مختلفة من البرنامج.

5. المصفوفات (القوائم)

# مثال على القوائم في بايثون

المصفوفات أو القوائم هي هياكل بيانات تُستخدم لتخزين مجموعات من القيم.

6. الكائنات (القواميس)

25. استخدام الدوال المتكررة (Recursive Functions)

# مثال على الدوال المتكررة في بايثون
def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

print("العدد التراكمي لـ 5 هو:", factorial(5))

الدوال المتكررة تستدعي نفسها لحل مشكلة أصغر. في المثال، نحسب القيمة التراكمية للعدد باستخدام الدالة `factorial` التي تستدعي نفسها حتى تصل إلى الحالة الأساسية.

27. استخدام قواعد البيانات SQLite

# مثال على استخدام قواعد البيانات SQLite في بايثون
import sqlite3

# الاتصال بقاعدة البيانات (أو إنشائها إذا لم تكن موجودة)
conn = sqlite3.connect("example.db")
cursor = conn.cursor()

# إنشاء جدول
cursor.execute('''CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)''')

# إدخال بيانات
cursor.execute("INSERT INTO users (name, age) VALUES ('علي', 28)")
conn.commit()

# استعلام البيانات
cursor.execute("SELECT * FROM users")
print("البيانات في قاعدة البيانات:", cursor.fetchall())

# إغلاق الاتصال
conn.close()

تستخدم مكتبة `sqlite3` للتعامل مع قواعد البيانات SQLite. في المثال، نخلق قاعدة بيانات، نضيف بيانات، ثم نقوم باستعلام البيانات.

data = json.loads(json_data)

36. استخدام `datetime` للتعامل مع التواريخ والأوقات

# مثال على استخدام مكتبة datetime للتعامل مع التواريخ والأوقات
from datetime import datetime

# الحصول على التاريخ والوقت الحالي
now = datetime.now()
print("الوقت الحالي:", now)

# تنسيق التاريخ والوقت

38. استخدام `sqlite3` للتعامل مع قواعد بيانات SQLite

# مثال على استخدام مكتبة sqlite3 للتعامل مع قواعد بيانات SQLite
import sqlite3

# إنشاء اتصال بقاعدة البيانات
conn = sqlite3.connect('example.db')
cursor = conn.cursor()

# إنشاء جدول
cursor.execute('''CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)''')

# إدخال بيانات
cursor.execute('''INSERT INTO users (name, age) VALUES (?, ?)''', ('سارة', 28))

# استعلام البيانات
cursor.execute('''SELECT * FROM users''')
rows = cursor.fetchall()
print("البيانات من قاعدة البيانات:")
for row in rows:
    print(row)

# إغلاق الاتصال
conn.close()

تستخدم مكتبة `sqlite3` للتعامل مع قواعد بيانات SQLite. في المثال، ننشئ جدولًا، ندخل بيانات، ونقوم باستعلام البيانات من القاعدة.
